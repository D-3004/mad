1. Open https://flutlab.io

5. Open lib/main.dart

6. Delete existing code

7. Paste the below code

import 'package:flutter/material.dart';

void main() {
  runApp(const MaterialApp(
    title: 'Flutter Navigation Demo',
    debugShowCheckedModeBanner: false,
    home: FirstRoute(),
  ));
}

class FirstRoute extends StatelessWidget {
  const FirstRoute({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('First Screen'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Image.network(
              'https://picsum.photos/250?image=9',
              height: 200,
            ),
            const SizedBox(height: 20),
            ElevatedButton(
              style: ElevatedButton.styleFrom(
                backgroundColor: Colors.orangeAccent,
              ),
              onPressed: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(
                    builder: (context) => const SecondRoute(),
                  ),
                );
              },
              child: const Text('Go to Second Screen'),
            ),
          ],
        ),
      ),
    );
  }
}

class SecondRoute extends StatelessWidget {
  const SecondRoute({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Second Screen'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Image.network(
              'https://picsum.photos/250?image=10',
              height: 200,
            ),
            const SizedBox(height: 20),
            ElevatedButton(
              style: ElevatedButton.styleFrom(
                backgroundColor: Colors.orangeAccent,
              ),
              onPressed: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(
                    builder: (context) => const ThirdRoute(),
                  ),
                );
              },
              child: const Text('Go to Third Screen'),
            ),
          ],
        ),
      ),
    );
  }
}

class ThirdRoute extends StatelessWidget {
  const ThirdRoute({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Third Screen'),
      ),
      body: Center(
        child: ElevatedButton(
          style: ElevatedButton.styleFrom(
            backgroundColor: Colors.blueGrey,
          ),
          onPressed: () {
            Navigator.pushAndRemoveUntil(
              context,
              MaterialPageRoute(
                builder: (context) => const FirstRoute(),
              ),
              (route) => false,
            );
          },
          child: const Text('Go Back to First Screen'),
        ),
      ),
    );
  }
}

8. Click "Run"

Output:
An app with three screens will be displayed where user can navigate between screens using buttons. 

Conclusion:
In this experiment, navigation and routing in Flutter app is implemented. 



Description:

In this experiment, navigation and routing mechanisms were implemented in a Flutter application using the FlutLab platform. The primary objective was to understand how multiple screens can be connected and how users can move seamlessly between them.

The application consists of three distinct screens, each designed using the Scaffold widget and containing UI elements such as images and buttons. Navigation between these screens is handled using Flutter’s built-in Navigator class along with MaterialPageRoute, which enables smooth transitions and route management.

The Navigator.push() method is used to move forward from one screen to another, demonstrating stack-based navigation where each new screen is placed on top of the previous one. Additionally, the Navigator.pushAndRemoveUntil() method is utilized to return to the initial screen while clearing the navigation stack, showcasing advanced routing control.

This experiment emphasizes how Flutter manages routes internally using a stack structure and highlights the importance of proper navigation handling in building multi-screen applications. It also demonstrates how user interaction triggers transitions between different views, enhancing the overall user experience.
